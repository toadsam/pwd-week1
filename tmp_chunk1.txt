    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const modalOverlay = document.getElementById('modal-overlay');
            const modalTitle = document.getElementById('modal-header-title');
            const modalBody = document.getElementById('modal-body');
            const closeModalBtn = document.getElementById('modal-close-btn');
            const sections = document.querySelectorAll('.content-section');

            // Function to open the modal
            const openModal = (title, contentHTML) => {
                modalTitle.textContent = title;
                // Safely inject content, removing any potential click listeners
                let tempDiv = document.createElement('div');
                tempDiv.innerHTML = contentHTML;
                const innerSections = tempDiv.querySelectorAll('.content-section');
                innerSections.forEach(s => s.classList.remove('content-section'));
                
                modalBody.innerHTML = tempDiv.innerHTML;
                modalOverlay.style.display = 'flex';
                document.body.style.overflow = 'hidden'; // Prevent background scrolling
            };

            // Function to close the modal
            const closeModal = () => {
                modalOverlay.style.display = 'none';
                document.body.style.overflow = 'auto'; // Restore background scrolling
                modalBody.innerHTML = '';
            };

            // Attach click listeners to all content sections
            sections.forEach(section => {
                section.addEventListener('click', (event) => {
                    const title = section.getAttribute('data-modal-title');
                    const contentSource = section.querySelector('.modal-body-source');
                    
                    if (title && contentSource) {
                        openModal(title, contentSource.innerHTML);
                    }
                });
            });

            // Attach close handlers
            closeModalBtn.addEventListener('click', closeModal);
            modalOverlay.addEventListener('click', (event) => {
                // Close modal when clicking outside the content box
                if (event.target === modalOverlay) {
                    closeModal();
                }
            });

            // Close modal with ESC key
            document.addEventListener('keydown', (event) => {
                if (event.key === 'Escape' && modalOverlay.style.display === 'flex') {
                    closeModal();
                }
            });
        });
    </script>
